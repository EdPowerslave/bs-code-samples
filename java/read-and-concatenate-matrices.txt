// Arayüzün/interface ne olduðunu ve ne iþe yaradýðýný biraz daha Javaya alýþtýkça öðreneceðinizi tahmin ediyorum. Ama genel programlama açýsýndan template gibi düþünebilirsiniz, özellikle burada daha sonra çýktý alacaðýmýz metodlarýn ne ile çalýþtýðýný ve ne sonuç verdiðini gösteriyorlar.

interface IMATH {
     public int add(int a, int b);
     public void printArray(int a[]);
     public void printArray(int a[][]);
     public int[][] sum(int a[][], int b[][]);
}

class Morpheus implements IMATH{
     public int add(int x, int y){ 
          return x + y;
     }

     public void printArray(int a[]){ // Tek boyutlu listeyi yazdýran metod
          for(int i=0; i<a.length; i++){ // Öðeleri dolaþýyor
               System.out.println(a[i]);
          }
     }

     public void printArray(int a[][]){ // 2 boyutlu listeyi yazdýran metod
          for(int r=0; r<a.length; r++){ // Satýrlarý dolaþýyor
               for(int c=0; c<a[0].length; c++){ // O satýrýn kolonlarýný dolaþýyor
                    System.out.println(a[r][c] + " ");
               }
          }
     }

     public int[][] sum(int[][] a, int[][] b){ // 2 tane boyutlarý ayný 2 boyutlu listeyi alýp onlarý toplayan metod
          int row = a.length; // Satýr sayýsý
          int column = a[0].length; // Kolon sayýsý
          int[][] basket= new int[row][column];

          for(int r=0; r<row; r++){ // Satýrlarý dolaþýyor
               for(int c=0; c<column; c++){ // O satýrýn kolonlarýný dolaþýyor
                    basket[r][c] = a[r][c] + b[r][c]; // Yeni matrise öðeleri yerleþtiriyor
               }
          } 
          return basket; // Yeni matrisi çaðýrýyor
     }
}
