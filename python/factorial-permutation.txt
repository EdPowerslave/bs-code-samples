def factorial(f):
  result = 1 
  if f < 0: 
    print ("Negatif sayýlar için faktöriyel bulunmaz.")
  elif f == 0: 
    print ("O ýn faktöriyel 1 dir.")
  else:
    for i in range(1,f + 1): 
      result = result*i 
  return result
  
print (factorial(12))

def permutation(x,y):
  resultp = 0
  if (y>x):
    resultp = resultp
  else:
    resultp = factorial(x)/factorial(x-y)
  return resultp
  
print (permutation(5,3))
